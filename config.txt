# --- Graphics & Display ---
# Use the open-source Full KMS driver (Required for Wayland/Sway/Hyprland)
dtoverlay=vc4-kms-v3d
# Limit GPU memory to save system RAM. 
# 64MB is the "sweet spot" for a 1080p Wayland session on 512MB total RAM.
gpu_mem=64

# --- CPU & Performance ---
# Enable 64-bit mode (The Zero 2 W supports arm64)
arm_64bit=1
# Minimal overclock (Optional: Zero 2 usually handles 1.2GHz safely)
# arm_freq=1200

# --- Hardware Interfaces ---
# Disable what you don't use to save power/interupts
dtparam=audio=on
dtparam=bluetooth=off  # Uncomment to disable BT
dtparam=spi=off        # Uncomment to disable SPI
dtparam=i2c_arm=off    # Uncomment to disable I2C

# --- Boot Behavior ---
# Disable the rainbow splash screen for a cleaner boot
disable_splash=1
# Use a standard UART for console if needed
enable_uart=1

# --- Camera / Display Auto-detect ---
camera_auto_detect=1
display_auto_detect=1
